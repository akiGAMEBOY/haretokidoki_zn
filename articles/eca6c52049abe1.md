```powershell:
$array = New-Object 'string[,]' 41,26
for ($i = 0; $i -lt $array.GetLength(0); $i++) {
    for ($j = 0; $j -lt $array.GetLength(1); $j++) {
        if ($j -eq 25) {
            $char = "AA" # 26番目の列は'AA'とする
        }
        else {
            $char = [char](66 + $j) # ASCIIコードで'B'から'Z'までを取得
        }

        $array[$i,$j] = "@${char}" + ($i + 2) # セルの参照を作成
    }
}
```
