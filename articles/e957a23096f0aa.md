function Test-FileExtension {
    param (
        [Parameter(Mandatory=$true)]
        [ValidateScript({ ($_ -ne $null) -and ($_.Trim() -ne '') })]
        [System.String]$PrintSring
    )

    Write-Host $PrintSring
}
